230  void process(const uint8_t *aMessage, size_t aLength)
231  {
232  const auto *header = reinterpret_cast<const Header *>(aMessage);
233 
234  if (header->receiverUID == nodeUID) {
235  switch (header->messageType) {
236  case MessageType::Ack: {
237  auto ackMsg = reinterpret_cast<const AckMessage *>(aMessage);
238  handleAck(ackMsg->payload.code);
239  } break;
240  case MessageType::Answer: {
241  auto answerMsg = reinterpret_cast<const AnwMessage *>(aMessage);
242  uint8_t returnCode = handleAnswer(
243  answerMsg->payload.request, &aMessage[sizeof(AnwMessage)], answerMsg->payload.dataSize);
244  sendAck(answerMsg->transmitUID, returnCode);
245  } break;
246  case MessageType::Command: {
247  auto cmdMsg = reinterpret_cast<const ComMessage *>(aMessage);
248  uint8_t returnCode = handleCommand(cmdMsg->payload.command, cmdMsg->payload.value);
249  sendAck(cmdMsg->transmitUID, returnCode);
250  } break;
251  case MessageType::Request: {
252  auto reqMsg = reinterpret_cast<const ReqMessage *>(aMessage);
253  bool isRequestProcessed = processRequest(reqMsg->transmitUID, reqMsg->payload.request, reqMsg->payload.answerDataSize);
254  if (!isRequestProcessed) {
255  sendAck(reqMsg->transmitUID, 0);
256  }
257  } break;
258  }
259  }
260  }
