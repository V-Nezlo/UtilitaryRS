43 void update(uint8_t *aData, size_t aLength)
44 {
45 size_t left = aLength;

47
48 while (left) {
49 left -= parser.update(static_cast<uint8_t *>(aData) + (aLength - left), left);

231 void process(const uint8_t *aMessage, size_t aLength)
232 {
233 const auto *header = reinterpret_cast<const Header *>(aMessage);
234
235 if (header->receiverUID == nodeUID) {
236 switch (header->messageType) {
237 case MessageType::Ack: {
238 const auto ackMsg = reinterpret_cast<const AckMessage *>(aMessage);
239 handleAck(ackMsg->payload.code);
240 } break;
241 case MessageType::Answer: {
242 const auto answerMsg = reinterpret_cast<const AnwMessage *>(aMessage);
243 uint8_t returnCode = handleAnswer(
244 answerMsg->payload.request, &aMessage[sizeof(AnwMessage)], answerMsg->payload.dataSize);
245 sendAck(answerMsg->transmitUID, returnCode);
246 } break;
247 case MessageType::Command: {
248 const auto cmdMsg = reinterpret_cast<const ComMessage *>(aMessage);
249 uint8_t returnCode = handleCommand(cmdMsg->payload.command, cmdMsg->payload.value);
250 sendAck(cmdMsg->transmitUID, returnCode);
251 } break;
252 case MessageType::Request: {
253 const auto reqMsg = reinterpret_cast<const ReqMessage *>(aMessage);
254 const auto isRequestProcessed = processRequest(reqMsg->transmitUID, reqMsg->payload.request, reqMsg->payload.answerDataSize);
255 if (!isRequestProcessed) {
256 sendAck(reqMsg->transmitUID, 0);
257 }
258 } break;
259 case MessageType::Probe: {
260 sendAck(header->transmitUID, 1);
261 } break;
262 }
263 }
264 }
